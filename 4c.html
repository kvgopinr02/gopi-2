Binary Search
Algorithm Steps:
  1. Input size and elements (should be sorted).
  2. Read key to search.
  3. Perform binary search (recursively):
    • Find the middle index.
    • If key = arr[mid], return position.
    • If key < arr[mid], search left half.
    • If key > arr[mid], search right half.
  4. If not found, return -1.
Python Code
def binary_search(arr, low, high, key):
    if low <= high:
        mid = (low + high) // 2
        if arr[mid] == key:
            return mid + 1
        elif arr[mid] > key:
            return binary_search(arr, low, mid - 1, key)
        else:
            return binary_search(arr, mid + 1, high, key)
    return -1

arr = list(map(int, input("Enter sorted array: ").split()))
key = int(input("Enter key to search: "))
result = binary_search(arr, 0, len(arr) - 1, key)
print("Found at position" if result != -1 else "Element not found", result if result != -1 else "")

#include <stdio.h>

void read(int arr[], int n) {
    for(int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
}

void display(int arr[], int n) {
    for(int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

int binary(int arr[], int low, int high, int key) {
    if(low <= high) {
        int mid = (low + high) / 2;
        if(arr[mid] == key)
            return mid + 1;
        else if(arr[mid] > key)
            return binary(arr, low, mid - 1, key);
        else
            return binary(arr, mid + 1, high, key);
    }
    return -1;
}

int main() {
    int n, key;
    printf("Enter size of array: ");
    scanf("%d", &n);
    int arr[n];
    read(arr, n);
    display(arr, n);
    printf("Element to search: ");
    scanf("%d", &key);
    int x = binary(arr, 0, n - 1, key);
    if(x == -1)
        printf("Element not found.");

}

Java Code
import java.util.*;

public class BinarySearch {
    public static int binary(int[] arr, int low, int high, int key) {
        if (low <= high) {
            int mid = (low + high) / 2;
            if (arr[mid] == key)
                return mid + 1;
            else if (arr[mid] > key)
                return binary(arr, low, mid - 1, key);
            else
                return binary(arr, mid + 1, high, key);
        }
        return -1;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter size of array: ");
        int n = sc.nextInt();
        int[] arr = new int[n];
        System.out.println("Enter sorted array elements:");
        for (int i = 0; i < n; i++) {
            arr[i] = sc.nextInt();
        }
        System.out.print("Enter key to search: ");
        int key = sc.nextInt();
        int pos = binary(arr, 0, n - 1, key);
        if (pos == -1)
            System.out.println("Element not found.");
        else
            System.out.println("Found at position " + pos);
    }
}
