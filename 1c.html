Tower of Hanoi
Algorithm Steps:
  1. Input the number of disks (n).
  2. Move the top n-1 disks from source rod to auxiliary rod.
  3. Move the nth disk to the destination rod.
  4. Move the n-1 disks from auxiliary rod to destination rod.
C Code
#include <stdio.h>

void hanoi(int n,char s,char d,char a){
    if (n==1){
        printf("Move disk %d from %c rod to %c rod\n",n,s,d);
    }
    else{
        hanoi(n-1,s,a,d);
        printf("Move disk %d from %c rod to %c rod\n",n,s,d);
        hanoi(n-1,a,d,s);
    }
}

int main(){
    int n;
    printf("Enter number of disks: ");
    scanf("%d",&n);
    hanoi(n,'a','c','b');
}

Java Code
import java.util.*;

public class hanoi {
    void hanoi(int n, char s, char d, char a) {
        if (n == 1) {
            System.out.println("Move disk " + n + " from " + s + " to " + d);
            return;
        }
        hanoi(n - 1, s, a, d);
        System.out.println("Move disk " + n + " from " + s + " to " + d);
        hanoi(n - 1, a, d, s);
    }

    public static void main(String args[]) {
        System.out.println("Enter number of disks: ");
        Scanner s = new Scanner(System.in);
        int x = s.nextInt();
        hanoi obj = new hanoi(); 
        obj.hanoi(x, 'a', 'c', 'b');
        s.close();
    }
}
