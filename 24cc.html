String Functions C Code

Algorithm

  1. Start the program.
  2. Prompt the user to enter two strings.
  3. Calculate the length of the first string.
  4. Copy the first string to another variable.
  5. Reverse the second string.
  6. Compare the two strings for equality and order.
  7. Reverse the second string again and concatenate it with the first string.
  8. Display all results.
  9. End the program.

C Code

#include <stdio.h>
#include <string.h>

int stringLength(const char *str) {
    int len = strlen(str);
    return len;
}

void stringCopy(char *dest, const char *src) {
    strcpy(dest, src);
}

void stringReverse(char *str) {
    int n = strlen(str);
    for (int i = 0; i < n / 2; i++) {
        char temp = str[i];
        str[i] = str[n - i - 1];
        str[n - 1 - i] = temp;
    }
}

int stringCompare(const char *str1, const char *str2) {
    return strcmp(str1, str2);
}

void stringConcatenate(char *dest, const char *src) {
    int length = stringLength(dest);
    int i = 0;
    while (src[i] != '\0') {
        dest[length + i] = src[i];
        i++;
    }
    dest[length + i] = '\0';
}

int main() {
    char str1[100], str2[50], result[150];

    printf("string1: ");
    scanf("%s", str1);
    printf("string2: ");
    scanf("%s", str2);

    int length = stringLength(str1);
    printf("Length of str1: %d\n", length);

    stringCopy(result, str1);
    printf("Copy of str1: %s\n", result);

    stringReverse(str2);
    printf("Reverse of str2: %s\n", str2);

    int cmp = stringCompare(str1, str2);
    if (cmp == 0) {
        printf("str1 and str2 are equal\n");
    } else if (cmp < 0) {
        printf("str1 is less than str2\n");
    } else {
        printf("str1 is greater than str2\n");
    }

    stringCopy(result, str1);
    stringReverse(str2);
    stringConcatenate(result, str2);
    printf("Concatenation of str1 and str2: %s\n", result);

    return 0;
}
