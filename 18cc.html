Matrix Transpose C Code

Algorithm

  1. Start the program.
  2. Prompt the user to enter the number of rows and columns for the matrix.
  3. Declare the matrix with the specified dimensions.
  4. Read the matrix elements using readMatrix().
  5. Display the original matrix using printMatrix().
  6. Transpose the matrix using transposeMatrix().
  7. Display the transposed matrix.
  8. End the program.

C Code

#include <stdio.h>

int rows, cols;

void readMatrix(int matrix[rows][cols]) {
    printf("Enter the elements:\n");
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }
}

void printMatrix(int matrix[rows][cols]) {
    printf("Matrix:\n");
    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            printf("%d ", matrix[i][j]);
        }
        printf("\n");
    }
}

void transposeMatrix(int matrix[rows][cols]) {
    int transposed[cols][rows];

    for(int i = 0; i < rows; i++) {
        for(int j = 0; j < cols; j++) {
            transposed[j][i] = matrix[i][j];
        }
    }

    printf("Transpose:\n");
    for(int i = 0; i < cols; i++) {
        for(int j = 0; j < rows; j++) {
            printf("%d ", transposed[i][j]);
        }
        printf("\n");
    }
}

int main() {
    printf("Enter number of rows: ");
    scanf("%d", &rows);
    printf("Enter number of columns: ");
    scanf("%d", &cols);
    
    int matrix[rows][cols];
    
    readMatrix(matrix);
    printMatrix(matrix);
    transposeMatrix(matrix);
    
    return 0;
}
