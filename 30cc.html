Pointer to Array and Pointers to Functions

Algorithm

  1. Start the program.
  2. Prompt the user to enter the size of the array and its elements.
  3. Create an array of pointers to store the addresses of the elements.
  4. Display the array elements using pointer dereferencing.
  5. Prompt the user for two integers.
  6. Use function pointers to perform addition and subtraction.
  7. Display the results of the operations.
  8. End the program.
C Code:

#include <stdio.h>
#include <stdlib.h>

int add(int x, int y);
int subtract(int x, int y);

int main() {
    int n;
    printf("size: ");
    scanf("%d", &n);
    printf("Elements: ");
    
    int *ptr3[n];
    int values[n];
    
    for(int i = 0; i < n; i++) {
        scanf("%d", &values[i]);
        ptr3[i] = &values[i];
    }

    printf("Accessing array elements using pointer: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", *ptr3[i]);
    }
    printf("\n");
    
    typedef int (*oper)(int, int);
    oper operation;
    int result, numA, numB;
    printf("Enter two integers: ");
    scanf("%d %d", &numA, &numB);
    operation = add;
    result = operation(numA, numB);
    printf("Using pointer to add: %d\n", result);
    
    operation = subtract;
    result = operation(numA, numB);
    printf("Using pointer to subtract: %d\n", result);
    
    return 0;
}

int add(int x, int y) {
    return x + y;
}

int subtract(int x, int y) {
    return x - y;
}
